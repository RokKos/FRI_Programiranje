FRAME label=_main
depth=1 size=32 locs=0 args=16
FP=T2 RV=T3
entryLabel=L2 exitLabel=L3 temps=0
% --- Prolog ---
_main
SET $0,16
% Storing FP
SUB $0,SP,$0
STO FP,$0,0
% STORING RA
GET $1,rJ
STO $1,$0,8
% Lowering FP
SET FP,SP
% Lowering SP
SET $0,32
SUB SP,SP,$0
JMP L2
L2
SET $0,0
SET $1,$0
SET $0,82
SET $0,$0
STO $1,$254,0
STO $0,$254,8
PUSHJ $8,_putChar
LDO $0,$254,0
SET $0,42
SET $0,$0
JMP L3
% --- Epilogue ---
L3
STO $0,FP,0 % Save return value
% Highering Stack pointer
SET SP,FP
% Getting RA
SET $0,16
SUB $0,SP,$0
LDO $1,$0,8
PUT rJ,$1
% Getting old FP
LDO FP,$0,0
POP 8,0
